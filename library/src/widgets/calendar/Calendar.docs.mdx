# Calendar Component

> A component that allows various selections of dates.

<Calendar
  onCalendarDayPress={undefined}
  calendarMonthId={toDateId(new Date())}
  calendarActiveDateRanges={[
    {
      startId: toDateId(new Date()),
      endId: toDateId(new Date()),
    },
  ]}
/>

```tsx twoslash
// @noErrors
import React from 'react';
// ---cut---
import {Calendar} from 'react-exo/calendar'; // [!code focus]

export default () => (
  <Calendar // [!code focus]
    onCalendarDayPress={undefined} // [!code focus]
    calendarMonthId={toDateId(new Date())} // [!code focus]
    calendarActiveDateRanges={[ // [!code focus]
      { // [!code focus]
        startId: toDateId(new Date()), // [!code focus]
        endId: toDateId(new Date()), // [!code focus]
      }, // [!code focus]
    ]} // [!code focus]
  /> // [!code focus]
);
```
