# Motion Layout

> A utility to declare animations to make it easy to transition between styles without needing to manage animations.

<Shift>
  {(value) => (
    <Motion.View
      initial={{x: 0, y: -50}}
      animate={{x: value * 100, y: 0}}
      whileHover={{scale: value * 1.5}}
      transition={{type: 'spring'}}
      whileTap={{y: 20}}
      style={{
        width: 100,
        height: 100,
        borderRadius: 10,
        backgroundColor: '#666',
      }}
    />
  )}
</Shift>

```tsx twoslash
import React from 'react';
import {Shift} from 'utils/blocks';
// ---cut---
// @log: ↓ Import the component
import {Motion} from 'react-exo/motion';

  animate={{x: 100, y: 0}}
  transition={{type: 'spring'}}
  whileHover={{scale: 1.5}}
  whileTap={{x: 20}}
/>
```
