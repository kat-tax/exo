# Form Service

> A service to handle form submission and validation.

```tsx twoslash
import React from 'react';
// ---cut---
// 1. Import the form utility // [!code focus]
import {Form} from 'react-exo/form'; // [!code focus]

// 2. Use the Form component // [!code focus]
export default () => (
  <Form // [!code focus]
    action="/api" // [!code focus]
    method="post" // [!code focus]
    onError={() => {}} // [!code focus]
    onSubmit={() => {}} // [!code focus]
    onSuccess={() => {}} // [!code focus]
    validateStatus={(status) => status >= 200} // [!code focus]
  /> // [!code focus]
);
```
