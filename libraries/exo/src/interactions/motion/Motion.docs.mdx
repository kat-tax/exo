# Motion Interaction

> A utility to declare animations to make it easy to transition between styles without needing to manage animations.

<Shift>
  {(value) => (
    <Motion.View
      initial={{x: 0, y: -50}}
      animate={{x: value * 100, y: 0}}
      whileHover={{scale: value * 1.5}}
      transition={{type: 'spring'}}
      whileTap={{y: 20}}
      style={{
        width: 100,
        height: 100,
        borderRadius: 10,
        backgroundColor: '#666',
      }}
    />
  )}
</Shift>

```tsx twoslash
import React from 'react';
import {Shift} from 'utils/blocks';
// ---cut---
// 1. Import the motion utility // [!code focus]
import {Motion} from 'react-exo/motion'; // [!code focus]

// 2. Try the Motion.View component // [!code focus]
export default () => (
  <Motion.View // [!code focus]
    initial={{x: 0, y: -50}} // [!code focus]
    animate={{x: 100, y: 0}} // [!code focus]
    transition={{type: 'spring'}} // [!code focus]
    whileHover={{scale: 1.5}} // [!code focus]
    whileTap={{x: 20}} // [!code focus]
  /> // [!code focus]
);
```
